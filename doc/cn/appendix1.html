App配置


// 配置路由和拦截器所在包（推荐配置）
Blade.defaultRoute("blade.sample.*");

// 配置路由和拦截器所在包（单独配置）
Blade.routes("blade.sample.route", "blade.sample.admin.route");
Blade.interceptor("blade.sample.interceptor");

// 配置IOC容器要管理的包
Blade.ioc("blade.sample.service.*");

// 配置视图位置和默认后缀名称
Blade.view("/WEB-INF/views/", ".jsp");

// 配置要过滤的静态目录
Blade.staticFolder("/assets/", "/upload/");
